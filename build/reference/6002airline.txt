@@example

//Classes which deal with flights and the airline
namespace Airline.flights;
class Airline{
  1 -- * RegularFlight;
}

class RegularFlight{
  Time time;
  unique Integer flightNumber;
  1 -- * SpecificFlight;
}

class SpecificFlight{
  unique Date date;
}

//Classes which deal with people
namespace Airline.humanResources;
class PassengerRole
{
  isA PersonRole;
  immutable String name ;
  1 -- * Booking;
}
  

class EmployeeRole
{
  String jobFunction ;
  isA PersonRole;
  * -- 0..1 EmployeeRole supervisor;
}

class Person
{
  settable String name;
  Integer idNumber;
  1 -- 0..2 PersonRole;
}

class PersonRole{}

class Booking{
  String seatNumber;
}

//Associations between subcomponents
association {
  * EmployeeRole -- * SpecificFlight;
}

association {
  * Booking -- 1 SpecificFlight;
}

association {
  1 Airline -- * Person;
}//$?[End_of_model]$?

class Person
{
  position 61 23 149 80;
  position.association Person__PersonRole 36,80 30,0;
}

class Airline
{
  position.association Airline__RegularFlight 25,46 30,0;
  position 266 24 109 45;
}

class PersonRole
{
  position 60 140 109 45;
}

class SpecificFlight
{
  position 370 286 109 45;
}

class RegularFlight
{
  position.association RegularFlight__SpecificFlight 48,63 50,0;
  position 358 159 109 63;
}

class PassengerRole
{
  position.association Booking__PassengerRole 55,63 11,0;
  position 165 259 113 63;
}

}
